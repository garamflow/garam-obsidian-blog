# 로그인 적용하기
## 1) 로그인 컴포넌트 구현
- 로그인을 수행할 컴포넌트를 만들어준다.
```javascript
import React, { useState } from 'react';
import { useDispatch } from 'react-redux';
import { login } from '../../slices/loginSlice';

const initState = {
  email: '',
  password: ''
}

const LoginComponent = () => {
  const [loginParam, setLoginParam] = useState({ ...initState });

  const dispatch = useDispatch();

  const handleChange = (e) => {
    loginParam[e.target.name] = e.target.value;
    setLoginParam({ ...loginParam });
  };

  const handleClickLogin = (e) => {
    dispatch(login(loginParam));
  };

  return (
    <div className="border-2 border-sky-200 mt-10 m-2 p-4">
      <div className="flex justify-center">
        <div className="text-4xl m-4 p-4 font-extrabold text-blue-500">Login Component</div>
      </div>
      <div className="flex justify-center">
        <div className="relative mb-4 flex w-full flex-wrap items-stretch">
          <div className="w-2/5 p-6 text-right font-bold">Email</div>
          <input className="w-1/5 p-6 rounded-r border border-solid border-neutral-500 shadow-md"
            name="email"
            type={'text'}
            value={loginParam.email}
            onChange={handleChange}
          />
        </div>
      </div>
      <div className="flex justify-center">
        <div className="relative mb-4 flex w-full flex-wrap items-stretch">
          <div className="w-2/5 p-6 text-right font-bold">Password</div>
          <input
            className="w-1/5 p-6 rounded-r border border-solid border-neutral-500 shadow-md"
            name="password"
            type={'password'}
            value={loginParam.password}
            onChange={handleChange}
          />
        </div>
      </div>
      <div className="flex justify-center">
        <div className="relative mb-4 flex w-full justify-center">
          <div className="w-2/5 p-6 flex justify-center font-bold">
            <button
              className="rounded p-4 w-36 bg-blue-500 text-xl text-white"
              onClick={handleClickLogin}
            >
              LOGIN
            </button>
          </div>
        </div>
      </div>
    </div>
  );
}

export default LoginComponent;
```

## 2) 로그인 리듀서 구현
- 로그인을 하면 상태값이 변하게 만들어줘야한다.
- 리듀서를 이용해서 상태를 바꿀 수 있다.
```javascript
import { createSlice } from '@reduxjs/toolkit';

const initState = {
  email: ''
}

const loginSlice = createSlice({
  name: 'loginSlice',
  initialState: initState,
  reducers: {
    login: (state, action) => {
      console.log('login');
      return { email: action.payload.email }
    },
    logout: () => {
      console.log('logout');
    }
  }
});

export const { login, logout } = loginSlice.actions;

export default loginSlice.reducer;
```
